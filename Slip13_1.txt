package com.example.slip13_1;

import androidx.appcompat.app.AppCompatActivity;

import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;
import android.widget.Toast;

public class MainActivity extends AppCompatActivity {
    EditText e1, e2, e3, e4;
    Button b1;
    String name, addr, no, age;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        e1 = findViewById(R.id.editTextTextPersonName);
        e2 = findViewById(R.id.editTextTextPersonName2);
        e3 = findViewById(R.id.editTextTextPersonName3);
        e4 = findViewById(R.id.editTextTextPersonName5);
        b1 = findViewById(R.id.button);


        b1.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                if(!validateName() | !validateAddr() | !validateNo() | !validateAge()){
                    return;
                }
                else{
                    Toast.makeText(MainActivity.this, "Registered successfully", Toast.LENGTH_SHORT).show();
                }

            }
        });


    }

    private boolean validateAge() {
        age=e4.getText().toString();
        int value=Integer.parseInt(age);
        String ageExp="(?=.*[0-9])"+".{2}";
        //int exp=Integer.parseInt(ageExp);
        if(age.isEmpty()){
            e4.setError("field is required");
            return false;
        }
        else if (!age.matches(ageExp) | (value<14)) {
            e4.setError("Invalid input");
            return false;
        } else {
            e4.setError(null);
            e4.setEnabled(false);
            return true;
        }

    }

    private boolean validateNo() {
        no=e3.getText().toString();
        String noExp="(?=.*[0-9])"+".{10}";
        if(no.isEmpty()){
            e3.setError("field is required");
            return false;
        }
        else if (!no.matches(noExp)){
            e3.setError("Invalid input");
            return false;
        }
        else{
            e3.setError(null);
            e3.setEnabled(false);
            return true;
        }
    }

    private boolean validateAddr() {
        addr=e2.getText().toString();
        if (addr.isEmpty()){
            e2.setError("field is required");
            return false;
        }
        else {
            e2.setError(null);
            e2.setEnabled(false);
            return true;
        }
    }

    private boolean validateName() {
        name=e1.getText().toString();
        String nameExp="[A-Za-z]+";
        if(name.isEmpty()){
            e1.setError("field is required");
            return false;
        }
        else if (!name.matches(nameExp)) {
            e1.setError("invalid input");
            return false;
        }
        else {
            e1.setError(null);
            e1.setEnabled(false);
            return true;
        }

    }
    }
